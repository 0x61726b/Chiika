AnimeListMixin = 
{
  name: null,
  ipcCall: function() {
    return this.setGrid();
  },
  setGridNew: function() {
    return this.grid = chiika.domManager.addNewGrid('anime', this.name, this.animeStatus);
  },
  setGrid: function() {
    var lastCol, lastColPos;
    this.setGridNew();
    return;
    chiika.appOptions.reorderColumns = false;
    this.grid = chiika.domManager.addNewGrid('anime', this.name, this.animeStatus);
    lastColPos = -1;
    lastCol = {};
    if (chiika.appOptions.reorderColumns) {
      w2ui[this.name].on('columnDragStart', (function(_this) {
        return function(e) {
          lastColPos = e.origColumnNumber;
          return lastCol = _this.grid.columns[lastColPos];
        };
      })(this));
      return w2ui[this.name].on('columnDragEnd', (function(_this) {
        return function(ex) {
          var findObj, findObjColumn, index, match, newColPos, timeOutFnc;
          _this.grid.columns = w2ui[_this.grid.name].columns;
          newColPos = ex.targetColumnNumber + 1;
          timeOutFnc = function(o) {
            return _this.findGridOrderAfterDrag(o);
          };
          setTimeout(timeOutFnc, 500, lastCol);
          return;
          if (lastColPos === newColPos) {
            return;
          }
          findObj = {};
          findObjColumn = {};
          findObj.column = findObjColumn;
          findObj.column.name = lastCol.field;
          match = _.find(chiika.appOptions.AnimeListColumns, findObj);
          index = _.indexOf(chiika.appOptions.AnimeListColumns, match);
          if (index === -1) {
            return chiika.logDebug("There is a problem dragging the column.");
          } else {
            match.column.order = newColPos;
            chiika.appOptions.AnimeListColumns.splice(index, 1, match);
            chiika.logDebug(lastCol.field + " dragged from " + lastColPos + " to " + newColPos);
            chiika.applicationDelegate.saveOptions(chiika.appOptions);
            return chiika.gridManager.prepareGridData(chiika.appOptions);
          }
        };
      })(this));
    }
  },
  findGridOrderAfterDrag: function(orig) {
    var counter, findObj, findObjColumn, grid, index, match, newColumn, newIndex;
    grid = w2ui[this.grid.name];
    newColumn = _.find(grid.columns, {
      field: orig.field
    });
    newIndex = _.indexOf(grid.columns, newColumn);
    findObj = {};
    findObjColumn = {};
    findObj.column = findObjColumn;
    findObj.column.name = orig.field;
    match = _.find(chiika.appOptions.AnimeListColumns, findObj);
    index = _.indexOf(chiika.appOptions.AnimeListColumns, match);
    if (index === -1) {
      return chiika.logDebug("There is a problem dragging the column.");
    } else {
      counter = 0;
      _.forEach(grid.columns, (function(_this) {
        return function(v, k) {
          var findCounterpartInOptionsArray, iFindCounterpartInOptionsArray, obj1, obj1Column;
          obj1 = {};
          obj1Column = {};
          obj1.column = obj1Column;
          obj1.column.name = v.field;
          findCounterpartInOptionsArray = _.find(chiika.appOptions.AnimeListColumns, obj1);
          iFindCounterpartInOptionsArray = _.indexOf(chiika.appOptions.AnimeListColumns, findCounterpartInOptionsArray);
          if (iFindCounterpartInOptionsArray !== -1) {
            findCounterpartInOptionsArray.column.order = counter;
            chiika.appOptions.AnimeListColumns.splice(iFindCounterpartInOptionsArray, 1, findCounterpartInOptionsArray);
          }
          return counter = counter + 1;
        };
      })(this));
      chiika.logDebug(orig.field + " dragged to " + newIndex);
      chiika.applicationDelegate.saveOptions(chiika.appOptions);
      return chiika.gridManager.prepareGridData(chiika.appOptions);
    }
  },
  componentDidMount: function() {
    chiika.ipcListeners.push(this);
    if (!window.chiika.isWaiting) {
      return this.setGrid();
    }
  },
  componentWillUnmount: function() {
    _.pull(chiika.ipcListeners, this);
    this.grid.clearAll();
    return this.grid = null;
  }
}
